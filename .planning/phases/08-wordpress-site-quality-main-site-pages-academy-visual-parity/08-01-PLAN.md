3. Create `ansible/roles/nginx_wordpress/tasks/wordpress-plugins-zip.yml` with these 5 tasks:

```yaml
---
# Nginx WordPress Role - Install plugins from local ZIP files
# Each item in nginx_wordpress_plugins_zip: {name: plugin-slug, file: filename.zip}

- name: Nginx WordPress | ZIP Plugins | Check if WordPress is installed
  ansible.builtin.stat:
    path: "{{ nginx_wordpress_web_root }}/wp-config.php"
  register: wp_zip_check
  tags: [nginx-wordpress, wordpress, plugins, zip]

- name: Nginx WordPress | ZIP Plugins | Create temp directory
  ansible.builtin.file:
    path: /tmp/wp-plugin-zips
    state: directory
    owner: "{{ nginx_wordpress_nginx_user }}"
    group: "{{ nginx_wordpress_nginx_group }}"
    mode: "0755"
  when: wp_zip_check.stat.exists
  tags: [nginx-wordpress, wordpress, plugins, zip]

- name: Nginx WordPress | ZIP Plugins | Copy ZIP files to server
  ansible.builtin.copy:
    src: "plugins/{{ item.file }}"
    dest: "/tmp/wp-plugin-zips/{{ item.file }}"
    owner: "{{ nginx_wordpress_nginx_user }}"
    group: "{{ nginx_wordpress_nginx_group }}"
    mode: "0644"
  loop: "{{ nginx_wordpress_plugins_zip }}"
  when: wp_zip_check.stat.exists
  tags: [nginx-wordpress, wordpress, plugins, zip, molecule-notest]

- name: Nginx WordPress | ZIP Plugins | Install and activate ZIP plugins
  ansible.builtin.shell: |
    set -euo pipefail
    if {{ nginx_wordpress_wpcli_path }} plugin is-installed {{ item.name }} --path={{ nginx_wordpress_web_root }}; then
      echo "SKIP: {{ item.name }} already installed"
    else
      {{ nginx_wordpress_wpcli_path }} plugin install /tmp/wp-plugin-zips/{{ item.file }} --activate --path={{ nginx_wordpress_web_root }}
    fi
  args:
    executable: /bin/bash
  become_user: "{{ nginx_wordpress_nginx_user }}"
  loop: "{{ nginx_wordpress_plugins_zip }}"
  register: zip_plugin_result
  changed_when: "'SKIP:' not in zip_plugin_result.stdout"
  when: wp_zip_check.stat.exists
  tags: [nginx-wordpress, wordpress, plugins, zip, molecule-notest]

- name: Nginx WordPress | ZIP Plugins | Remove temp directory
  ansible.builtin.file:
    path: /tmp/wp-plugin-zips
    state: absent
  when: wp_zip_check.stat.exists
  tags: [nginx-wordpress, wordpress, plugins, zip]
```
