Add a service health check play to `ansible/playbooks/validate.yml` BEFORE the existing "Validate complete deployment" play (insert as the first play in the file). This verifies that critical services are running after a site.yml run, serving as an idempotency smoke test:

```yaml
- name: Validate critical service health
  hosts: wordpress_servers
  become: true
  gather_facts: false

  tasks:
    - name: Gather service facts
      ansible.builtin.service_facts:
      tags: [validate, health]

    - name: Assert nginx is running
      ansible.builtin.assert:
        that:
          - ansible_facts.services['nginx.service'].state == 'running'
        fail_msg: "nginx is not running"
        success_msg: "nginx is running"
      tags: [validate, health]

    - name: Assert MariaDB is running
      ansible.builtin.assert:
        that:
          - ansible_facts.services['mariadb.service'].state == 'running'
        fail_msg: "MariaDB is not running"
        success_msg: "MariaDB is running"
      tags: [validate, health]

    - name: Assert PHP-FPM main pool is running
      ansible.builtin.assert:
        that:
          - ansible_facts.services['php-fpm-main.service'] is defined
          - ansible_facts.services['php-fpm-main.service'].state == 'running'
        fail_msg: "PHP-FPM main pool is not running"
        success_msg: "PHP-FPM main pool is running"
      tags: [validate, health]
      failed_when: false

    - name: Assert Valkey is running
      ansible.builtin.assert:
        that:
          - ansible_facts.services['valkey.service'].state == 'running'
            or ansible_facts.services['valkey-server.service'].state == 'running'
        fail_msg: "Valkey is not running"
        success_msg: "Valkey is running"
      tags: [validate, health]
      failed_when: false
```
