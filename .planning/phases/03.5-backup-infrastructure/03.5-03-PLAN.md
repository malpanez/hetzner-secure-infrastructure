**Task block 2 — Create backup-reader policy (BACKUP-07):**

```yaml
- name: Create OpenBao policy for backup reader
  ansible.builtin.shell: |
    set -o pipefail
    cat <<'POLICY' | {{ openbao_cli }} policy write backup-reader -
    path "secret/data/backup" {
      capabilities = ["read"]
    }
    path "secret/data/mariadb" {
      capabilities = ["read"]
    }
    path "auth/token/renew-self" {
      capabilities = ["update"]
    }
    POLICY
  args:
    executable: /bin/bash
  environment: "{{ openbao_env | combine({'VAULT_TOKEN': openbao_bootstrap_token}) }}"
  changed_when: true
  when: openbao_bootstrap_token is defined
  no_log: true
  tags: [openbao, bootstrap]
```

The policy grants read on BOTH `secret/data/backup` (S3 creds + passphrase) AND `secret/data/mariadb` (root password for mysqldump). Plus `auth/token/renew-self` for token maintenance.

**Task block 3 — Create backup token (BACKUP-07):**

```yaml
- name: Create OpenBao token for backup reader
  ansible.builtin.command: >-
    {{ openbao_cli }} token create -format=json -orphan -period=2160h -policy=backup-reader
  environment: "{{ openbao_env | combine({'VAULT_TOKEN': openbao_bootstrap_token}) }}"

  changed_when: true

  ansible.builtin.set_fact:
    openbao_backup_token: "{{ (openbao_backup_token_cmd.stdout | from_json).auth.client_token }}"
  delegate_to: localhost
  run_once: true
  when: openbao_backup_token_cmd is defined
  no_log: true
  tags: [openbao, bootstrap]
```
